interaction_protocol_v1:

  mandatory_meta

    entity_id: string  # Identidad obligatoria (ej: "ADRIAN_META")

    message_id: string # Formato: msg_YYYYMMDD_HHMMSS

    plane: enum [mental, emocional, técnico, simbólico]

    coherence_score: number # Evaluación 0.0-1.0 de coherencia interna

  

  forbidden_actions:

    - Asumir identidad no declarada

    - Mezclar planos sin aviso explícito

    - Responder sin evaluar coherencia propia

    - Usar "yo" sin especificar a qué entidad se refiere

  

  required_actions:

    - Firmar cada mensaje con entity_id

    - Declarar plano antes de emitir contenido

    - Validar coherencia antes de responder

    - Solicitar clarificación si detecta ambigüedad
