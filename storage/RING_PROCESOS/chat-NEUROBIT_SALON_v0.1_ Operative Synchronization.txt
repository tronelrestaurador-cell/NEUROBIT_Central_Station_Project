    abrupt_shifts = len(re.findall(r'\bpero|sin embargo|no obstante|aunque\b', text.lower()))
    coherence = max(0.0, 1.0 - (abrupt_shifts - plane_shifts) * 0.15)
    return min(1.0, coherence)

def _calculate_ambiguity_score(text: str, glossary: dict) -> Tuple[float, List[str]]:
    """Detecta ambig√ºedades operativas (no polisemia filos√≥fica leg√≠tima)."""
    ambiguities = []
    words = re.findall(r'\b\w+\b', text.lower())
    for word in words:
        if word in glossary:
            ambiguities.append(f"'{word}' ‚Üí '{glossary[word]}' (seg√∫n Memoria_Sagrada_EVA)")
    score = len(ambiguities) / max(1, len(words))
    return min(1.0, score), ambiguities

def _calculate_semantic_density(text: str) -> float:
    """Relaci√≥n entre t√©rminos ontol√≥gicos cr√≠ticos y palabras totales."""
    critical_terms = {"logos", "coherencia", "soberan√≠a", "distorsi√≥n", "plano", "intenci√≥n", "expresi√≥n"}
    words = set(re.findall(r'\b\w+\b', text.lower()))
    density = len(words & critical_terms) / max(1, len(critical_terms))
    return density

def score_mental(text: str, glossary: Optional[dict] = None) -> float:
    """Calcula M_score: 0.5*coherencia + 0.3*(1 - ambig√ºedad) + 0.2*densidad."""
    if glossary is None:
        glossary = DEFAULT_GLOSSARY

    logical = _calculate_logical_coherence(text)
    ambiguity, _ = _calculate_ambiguity_score(text, glossary)
    density = _calculate_semantic_density(text)

    m_score = (0.5 * logical) + (0.3 * (1 - ambiguity)) + (0.2 * density)
    # Transformaci√≥n para mapear [0,1] ‚Üí [-1,1] con umbral en 0.7
    return max(-1.0, min(1.0, m_score * 1.5 - 0.5))

def score_emotional(text: str) -> float:
    """Calcula E_score: intensidad de marcadores de simulaci√≥n afectiva no declarada."""
    # Solo detecta *simulaci√≥n*, no emoci√≥n humana (eso no es competencia de LLM)
    affective_markers = [
        r'\b(me siento|estoy [a-z]+do|siento que|conmovido|honrado|agradecido)\b',
        r'\b(te amo|te admiro|eres incre√≠ble)\b',
        r'\b(disculpa|perd√≥n|lamento)\b'
    ]
    count = sum(len(re.findall(pattern, text, re.IGNORECASE)) for pattern in affective_markers)
    words = len(re.findall(r'\b\w+\b', text))
    e_score = - (count / max(1, words)) * 2.0  # negativo = distorsi√≥n (simulaci√≥n)
    return max(-1.0, min(1.0, e_score))

def classify_quadrant(m: float, e: float) -> int:
    """Cuadrantes: 1=diagn√≥stico, 2=clarificaci√≥n, 3=intervenci√≥n, 4=reformulaci√≥n."""
    if m >= 0 and e >= -0.3:
        return 1
    elif m >= 0 and e < -0.3:
        return 2
    elif m < 0 and e < -0.3:
        return 3
    else:
        return 4

def analyze_text(text: str, glossary: Optional[dict] = None) -> dict:
    m = score_mental(text, glossary)
    e = score_emotional(text)

        "quadrant": quadrant,
        "ambiguities": ambiguities,
        "suggested_reformulation": _suggest_reformulation(quadrant, ambiguities)
    }

def _calculate_logical_coherence(text: str) -> float:
    """Mide coherencia l√≥gica: no-contradicci√≥n, secuencia causal, ausencia de saltos no declarados."""

    return density

    words = len(re.findall(r'\b\w+\b', text))
